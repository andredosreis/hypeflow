- Variáveis de ambiente de teste isoladas em `.env.test.local`.

## Technical Notes

### Estratégia de mock para route handlers (Next.js App Router)

Testar route handlers no App Router com Vitest requer mock das dependências, não execução do servidor:

```ts
// __tests__/api/ai/agent.test.ts
import { vi, describe, it, expect, beforeAll, afterAll, afterEach } from 'vitest'
import { POST } from '@/app/api/ai/agent/route'
import { NextRequest } from 'next/server'
import { server } from '@/mocks/server'

// Mock auth wrapper
vi.mock('@/lib/api/with-session', () => ({
  requireSession: vi.fn(),
  getClientIp: vi.fn().mockReturnValue('127.0.0.1'),
}))

// Mock rate limiter
vi.mock('@/lib/api/rate-limit', () => ({
  rateLimit: vi.fn().mockResolvedValue({ allowed: true, retryAfter: 0 }),
}))

beforeAll(() => server.listen({ onUnhandledRequest: 'error' }))
afterEach(() => server.resetHandlers())
afterAll(() => server.close())

describe('POST /api/ai/agent', () => {
  it('returns 401 when no session', async () => {
    const { requireSession } = await import('@/lib/api/with-session')
    vi.mocked(requireSession).mockResolvedValueOnce({
      response: new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401 }),
      user: null,
      supabase: null,
    })

    const req = new NextRequest('http://localhost/api/ai/agent', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Content-Length': '50' },
      body: JSON.stringify({ messages: [{ role: 'user', content: 'hi' }] }),
    })

    const res = await POST(req)
    expect(res.status).toBe(401)
  })

  it('returns 413 when body exceeds 16KB', async () => {
    const req = new NextRequest('http://localhost/api/ai/agent', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Content-Length': '20000' },
      body: JSON.stringify({ messages: [{ role: 'user', content: 'x'.repeat(17000) }] }),
    })

    const res = await POST(req)
    expect(res.status).toBe(413)
  })
})
```
